Line + area

A simple line and area chart.

import LayerCake from 'layercake';
import points from '../../data/points.js';
import Line from '../Line.html';
import Area from '../Area.html';
import AxisX from '../AxisX.html';
import AxisY from '../AxisY.html';

const myCake = new LayerCake({
  x: 'myX',
  y: d => d['myY'],
  yDomain: [0, null],
  data: points,
  target: document.getElementById('my-chart')
})
  .svgLayers([
    { component: AxisX, opts: {} },
    { component: AxisY, opts: {} },
    { component: Line, opts: {} },
    { component: Area, opts: {} }
  ]);

myCake.render();
<path class='path-line' d='{path}'></path>
<style>
  .path-line {
    fill: none;
    stroke: #ab00d6;
    stroke-linejoin: round;
    stroke-linecap: round;
    stroke-width: 2;
  }
</style>

<script>
export default {
  namespace: 'svg',
  computed: {
    path: ({ $data, $xGet, $yGet }) => {
      return 'M' + $data
        .map(function (d, i) {
          return $xGet(d) + ',' + $yGet(d);
        })
        .join('L');
    }
  }
};
</script>
<path class='path-area' d='{area}'></path>
<style>
  .path-area {
    fill: #ab00d610;
  }
</style>
<script>
export default {
  namespace: 'svg',
  computed: {
    area: ({ $xScale, $domains, $yScale, path }) => {
      const yRange = $yScale.range();
      return path + (
        'L' + $xScale($domains.x[1]) + ',' + yRange[0] +
        'L' + $xScale($domains.x[0]) + ',' + yRange[0] +
        'Z'
      );
    },
    path: ({ $data, $xGet, $yGet }) => {
      return 'M' + $data
        .map(function (d, i) {
          return $xGet(d) + ',' + $yGet(d);
        })
        .join('L');
    }
  }
};
</script>
<g class='axis x-axis'>
  {#each ticks as tick, i}
    <g class='tick tick-{ tick }' transform='translate({$xScale(tick)},{$yScale.range()[0]})'>
      {#if opts.gridlines !== false}
        <line y1='{$height * -1}' y2='0' x1='0' x2='0'></line>
      {/if}
      <text y='16' text-anchor='{i === 0 ? 'start' : i === ticks.length - 1 ? 'end' : 'middle'}'>{opts.formatTick ? opts.formatTick(tick) : tick}</text>
    </g>
  {/each}
  {#if opts.baseline === true}
    <line class="baseline" y1='{$height - 0.5}' y2='{$height - 0.5}' x1='0' x2='{$width - 1}'></line>
  {/if}
</g>

<script>
export default {
  namespace: 'svg',
  computed: {
    ticks: ({ $xScale, opts }) => {
      return $xScale.ticks(opts.tickNumber);
    }
  }
};
</script>
<g class='axis y-axis' transform='translate(-{$padding.left}, 0)'>
  {#each $yScale.ticks(opts.tickNumber || 5) as tick, i}
    <g class='tick tick-{tick}' transform='translate(0, {$yScale(tick)})'>
      {#if opts.gridlines !== false}
        <line x2='100%'></line>
      {/if}
      <text y='-4'>{opts.formatTick ? opts.formatTick(tick) : tick}</text>
    </g>
  {/each}
</g>
